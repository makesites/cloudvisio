Per capita income

countries split in high/medium/low groups

Data from worldbank.org

	vis = new Cloudvisio();

	d3.json("data/income.json", function( data ){

		vis
		.data( data[1] )
		.set({ layout: "force" })
		.axis("radius", 5)
		.axis("name")
		.select("incomeLevel")
		.group(["High", "Middle", "Low"]);

		vis.render();
	});
